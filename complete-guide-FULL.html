1. createDirectory()
Create directory if it doesn't exist (cross-platform with error handling)
bool createDirectory(const string &path) {
#ifdef _WIN32
    int result = _mkdir(path.c_str());
    return (result == 0) || (errno == EEXIST);
#else
    int result = mkdir(path.c_str(), 0755);
    return (result == 0) || (errno == EEXIST);
#endif
}
Windows vs Linux:
‚Ä¢ Windows: _mkdir() - no permissions parameter
‚Ä¢ Linux/Mac: mkdir() - requires permissions (0755 = rwxr-xr-x)
Return Value Logic:
‚Ä¢ result == 0 ‚Üí Directory created successfully
‚Ä¢ errno == EEXIST ‚Üí Directory already exists (OK!)
‚Ä¢ Returns false only if creation failed for other reasons
2. getTimestampForFilename()
Convert time format for safe filenames (parse and rebuild)
string getTimestampForFilename(const string &timeSlot) {
    // Parse time from "HH:MM AM/PM" format
    stringstream ss(timeSlot);
    int hour, minute;
    char colon;
    string period;
    ss >> hour >> colon >> minute >> period;
    
    // Rebuild with dashes instead of colons
    stringstream result;
    result << setfill('0') << setw(2) << hour << "-"
           << setfill('0') << setw(2) << minute << "-" << period;
    return result.str();
}
Examples:
"10:40 AM" ‚Üí "10-40-AM"
"01:50 PM" ‚Üí "01-50-PM"
"11:30 AM" ‚Üí "11-30-AM"
Why Parse Instead of Replace?
‚Ä¢ Ensures zero-padding ("8:30" becomes "08-30")
‚Ä¢ Validates format (parsing fails on invalid input)
‚Ä¢ Windows filenames cannot contain : character

4. formatTimeWithZeroPadding()
Parse and format time string with zero-padded hours
string formatTimeWithZeroPadding(const string &timeStr) {
    // Parse time from "H:MM AM/PM" or "HH:MM AM/PM" format
    stringstream ss(timeStr);
    int hour, minute;
    char colon;
    string period;
    ss >> hour >> colon >> minute >> period;
    
    // Format with zero-padded hour
    stringstream result;
    result << setfill('0') << setw(2) << hour << ":"
           << setfill('0') << setw(2) << minute << " " << period;
    return result.str();
}
Examples:

"12:20 PM" ‚Üí "12:20 PM"
"1:50 PM" ‚Üí "01:50 PM"
Why Parse? Input can have single-digit hours ("8:30"), output must have two digits ("08:30")
5. centerAlign()
Center text within specified width with left and right padding
string centerAlign(const string &text, int width) {
    int padding = width - text.length();
    if (padding <= 0) return text;  // Text too long, no padding
    
    int leftPad = padding / 2;
    int rightPad = padding - leftPad;  // Handle odd widths
    
    return string(leftPad, ' ') + text + string(rightPad, ' ');
}
Examples:
centerAlign("CSE-1121", 12) ‚Üí " CSE-1121 " (1 left, 2 right)
centerAlign("SL", 5) ‚Üí " SL " (1 left, 2 right)
centerAlign("Name", 10) ‚Üí " Name " (3 left, 3 right)
Why Both Pads? For odd-width padding, need different left/right counts
Example: "Hi" in width 5 = 3 spaces total ‚Üí 1 left + 2 right

10. getPeriodNumber()
Calculate period number using set for uniqueness
int getPeriodNumber(const string &courseCode, const string &date,
                    const string &timeSlot) {
    set uniqueTimeslots;
    
    // Collect all unique timeslots for this course on this date 
    // that come before or equal to current timeslot
    for (const auto &record : attendanceRecords) {
        if (record.courseCode == courseCode && record.date == date &&
            record.timeSlot <= timeSlot) {
            uniqueTimeslots.insert(record.timeSlot);
        }
    }
    
    // Count this timeslot's position
    int position = 0;
    for (const auto &ts : uniqueTimeslots) {
        position++;
        if (ts == timeSlot) return position;
    }
    
    // If not found in records yet, count it as next period
    return uniqueTimeslots.size() + 1;
}
Why use set? Automatically handles duplicates and keeps sorted order

12. getDayOfWeek()
Calculate day using Zeller's Congruence
int getDayOfWeek(int day, int month, int year) {
    if (month < 3) {
        month += 12;
        year--;
    }
    
    int k = year % 100;
    int j = year / 100;
    
    int h = (day + (13 * (month + 1)) / 5 + k + k/4 + j/4 - 2*j) % 7;
    
    return h;  // 0=Sat, 1=Sun, 2=Mon, 3=Tue, 4=Wed, 5=Thu, 6=Fri
}
Example: getDayOfWeek(27, 11, 2025) = 5 (Thursday)

14. isDateInFuture()
Check if date is after today
bool isDateInFuture(const string &dateStr) {
    int day, month, year;
    char sep1, sep2;
    stringstream ss(dateStr);
    ss >> day >> sep1 >> month >> sep2 >> year;
    
    if (sep1 != '-' || sep2 != '-') return true;
    
    time_t now = time(0);
    tm *ltm = localtime(&now);
    int currentDay = ltm->tm_mday;
    int currentMonth = 1 + ltm->tm_mon;
    int currentYear = 1900 + ltm->tm_year;
    
    if (year > currentYear) return true;
    if (year < currentYear) return false;
    if (month > currentMonth) return true;
    if (month < currentMonth) return false;
    if (day > currentDay) return true;
    return false;
}
15. validateDayDateMatch()
Verify date actually falls on selected day
bool validateDayDateMatch(const string &selectedDay, const string &dateStr) {
    int day, month, year;
    char sep1, sep2;
    stringstream ss(dateStr);
    ss >> day >> sep1 >> month >> sep2 >> year;
    
    if (sep1 != '-' || sep2 != '-') return false;
    
    int dayOfWeek = getDayOfWeek(day, month, year);
    string actualDay = getDayName(dayOfWeek);
    
    return (actualDay == selectedDay);
}
Why: Prevents user errors when entering dates

20. countClassesHeld()
Count unique class sessions actually held for a course
int countClassesHeld(const string &courseCode) {
    map uniqueClasses;
    
    for (const auto &record : attendanceRecords) {
        if (record.courseCode == courseCode) {
            // Create unique key: date + timeslot
            uniqueClasses[record.date + "_" + record.timeSlot] = 1;
        }
    }
    
    return uniqueClasses.size();
}
Example:
If EEE-1121 had attendance taken on:
‚Ä¢ 27-11-2025 at 11:30 AM
‚Ä¢ 28-11-2025 at 11:30 AM
‚Ä¢ 28-11-2025 at 2:40 PM
Result: countClassesHeld("EEE-1121") = 3
Why use map?
‚Ä¢ Map automatically handles duplicates (same key = one entry)
‚Ä¢ Combining date + timeslot creates unique key
‚Ä¢ Even if 45 students have records for same period, counts as 1 class
‚ö†Ô∏è Important Distinction:
‚Ä¢ countTotalClasses() = Semester total (48 or 32)
‚Ä¢ countClassesHeld() = Actually conducted (0-48/32)
21. calculateAttendance()
Calculate attendance percentage for student in course
double calculateAttendance(const string &studentId, const string &courseCode) {
    int attended = countAttendedClasses(studentId, courseCode);
    int total = countTotalClasses(courseCode);
    
    if (total == 0) return 0.0;
    
    double percentage = (static_cast(attended) / total) * 100.0;
    
    // Cap at 100% (edge case: more attended than total)
    if (percentage > 100.0) percentage = 100.0;
    
    return percentage;
}
Example Calculation: 
// Student C253027 in EEE-1121
int attended = 40;  // countAttendedClasses returns 40
int total = 48;     // countTotalClasses returns 48
double percentage = (40.0 / 48.0) * 100.0 = 83.33%
Why static_cast<double>?
Without cast: 40 / 48 = 0 (integer division)
With cast: 40.0 / 48.0 = 0.8333... (floating-point division)
‚ö†Ô∏è Dis-collegiate Threshold:
‚Ä¢ percentage >= 60.0 = Collegiate (GREEN)
‚Ä¢ percentage < 60.0 = Dis-collegiate (RED, at risk)
‚Ä¢ Threshold defined as: const double DISCOLLEGIATE_THRESHOLD = 60.0;
