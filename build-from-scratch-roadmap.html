// Add these BEFORE main()
#include <string>
#include <vector>
using namespace std;

struct Student {
    int sl;          // Serial number
    string id;       // Student ID (e.g., "C251027")
    string name;     // Full name
    Student(int s, string i, string n) : sl(s), id(i), name(n) {}
};

struct Course {
    string code;                // Course code (e.g., "EEE-1121")
    string title;               // Course title
    int weeklyClasses;          // Classes per week
    int totalSemesterClasses;   // Total classes in semester
    Course(string c = "", string t = "", int w = 0, int total = 48)
        : code(c), title(t), weeklyClasses(w), totalSemesterClasses(total) {}
};

struct ClassPeriod {
    string day;         // Day name (e.g., "Saturday")
    string time;        // Start time (e.g., "11:30 AM")
    string endTime;     // End time (e.g., "12:20 PM")
    string courseCode;  // Course code
    ClassPeriod(string d, string t, string et, string c)
        : day(d), time(t), endTime(et), courseCode(c) {}
};

struct AttendanceRecord {
    string studentId;   // Student ID
    string courseCode;  // Course code
    string date;        // Date (DD-MM-YYYY)
    string day;         // Day name
    string timeSlot;    // Time slot
    bool present;       // True = Present, False = Absent
    AttendanceRecord(string sid, string cc, string dt, string dy, string ts, bool p)
        : studentId(sid), courseCode(cc), date(dt), day(dy), timeSlot(ts), present(p) {}
};
